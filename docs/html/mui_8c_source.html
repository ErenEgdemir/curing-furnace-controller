97static size_t mui_fds_get_cmd_size_without_text(fds_t *s) MUI_NOINLINE;
98static size_t mui_fds_get_cmd_size_without_text(fds_t *s)
99{
100 uint8_t c = mui_get_fds_char(s);
101 c &= 0xdf; /* consider upper and lower case */
102 switch(c)
103 {
104 case 'U': return 2; // User Form: CMD (1 Byte), Form-Id (1 Byte)
105 case 'S': return 2; // Style: CMD (1 Byte), Style Id (1 Byte)
106 case 'D': return 3; // Data within Text: CMD (1 Byte), ID (2 Bytes), Text (does not count here)
107 case 'Z': return 3; // Zero field without x, y, arg & text: CMD (1 Byte), ID (2 Bytes)
108 case 'F': return 5; // Field without arg & text: CMD (1 Byte), ID (2 Bytes), X, Y
109 case 'B': return 5; // Field with text: CMD (1 Byte), ID (2 Bytes), X, Y, Text (does not count here)
110 case 'T': return 6; // Field with arg & text: CMD (1 Byte), ID (2 Bytes), X, Y, Arg, Text (does not count here)
111 case 'A': return 6; // Field with arg (no text): CMD (1 Byte), ID (2 Bytes), X, Y, Arg, Text
112 case 'L': return 3; // Text Label: CMD (1 Byte), X, Y (same as 'B' but with fixed ID '.L', MUIF_LABEL, MUI_LABEL)
113 case 'G': return 4; // Goto Btutton: CMD (1Byte), X, Y, Arg, Text (same as 'T' but with fixed ID '.G', MUIF_GOTO, MUI_GOTO)
114 case 0: return 0;
115 }
116 return 1;
117}

121/*
122 s must point to the string delimiter start: first '/' for "B00ab/ok/"
123 - '/' actually is 0xff
124 - return the total size of the string, including the delimiter
125 - copies the content of the string ("ok") to the ui text buffer
126
127*/
128static size_t mui_fds_parse_text(mui_t *ui, fds_t *s)
129{
130 uint8_t i = 0;
131 ui->delimiter = mui_get_fds_char(s);
132 uint8_t c;
133 fds_t *t = s;
134
135 //printf("mui_fds_parse_text del=%d\n", delimiter);
136#ifdef MUI_CHECK_EOFDS
137 if ( ui->delimiter == 0 )
138 return 0;
139#endif
140 t++;
141 for( ;; )
142 {
143 c = mui_get_fds_char(t);
144 //printf("mui_fds_parse_text i=%d, c=%c\n", i, c);
145#ifdef MUI_CHECK_EOFDS
146 if ( c == 0 )
147 break;
148#endif
149 if ( c == ui->delimiter )
150 {
151 t++;
152 break;
153 }
154 if ( i < MUI_MAX_TEXT_LEN )
155 {
156 ui->text[i++] = c;
157 }
158 t++;
159 }
160 ui->text[i] = '\0' ;
161 return t-s;
162}
163
164/*
165 get the first token within a text argument.
166 The text argument may look like this:
167 "B00ab/banana|apple|peach|cherry/"
168 The outer delimiter "/" is not fixed and can be any char except "|" and "\0"
169 The inner delimiter "|" is fixed. It must be the pipe symbol.
170 This function will place "banana" into ui->text if the result is not 0
171
172 if ( mui_fds_first_token(ui) )
173 {
174 do
175 {
176 // handle token in ui->text
177 } while ( mui_fds_next_token(ui) )
178 }
179
180*/
181
182uint8_t mui_fds_first_token(mui_t *ui)
183{
184 ui->token = ui->fds;
185 ui->token += mui_fds_get_cmd_size_without_text(ui->fds);
186 ui->delimiter = mui_get_fds_char(ui->token);
187 ui->token++; // place ui->token on the first char of the token
188 return mui_fds_next_token(ui);
189}
190
191/*
192 The inner token delimiter "|" is fixed. It must be the pipe symbol.
193*/
194uint8_t mui_fds_next_token(mui_t *ui)
195{
196 uint8_t c;
197 uint8_t i = 0;
198 // printf("mui_fds_next_token: call, ui->token=%p\n", ui->token);
199 for( ;; )
200 {
201 c = mui_get_fds_char(ui->token);
202 // printf("mui_fds_next_token: i=%d c=%c\n", i, c);
203#ifdef MUI_CHECK_EOFDS
204 if ( c == 0 )
205 break;
206#endif
207 if ( c == ui->delimiter )
208 break;
209 if ( c == '|' )
210 {
211 ui->token++; // place ui->token on the first char of the next token
212 break;
213 }
214
215 if ( i < MUI_MAX_TEXT_LEN )
216 {
217 ui->text[i++] = c;
218 }
219
220 ui->token++;
221 }
222 ui->text[i] = '\0' ;
223 if ( i == 0 )
224 return 0; // no further token found
225 return 1; // token placed in ui->text
226}

228/*
229 find nth token ('|' delimiter), return 0 if n exceeds the number of tokens, 1 otherwise
230 the result is stored in ui->text
231*/
232uint8_t mui_fds_get_nth_token(mui_t *ui, uint8_t n)
233{
234 // printf("mui_fds_get_nth_token: call, n=%d\n", n);
235 if ( mui_fds_first_token(ui) )
236 {
237 do
238 {
239 if ( n == 0 )
240 {
241 // printf("mui_fds_get_nth_token: found");
242 return 1;
243 }
244 n--;
245 } while ( mui_fds_next_token(ui) );
246 }
247 //printf("mui_fds_get_nth_token: NOT found\n");
248 return 0;
249}

251uint8_t mui_fds_get_token_cnt(mui_t *ui)
252{
253 uint8_t n = 0;
254 if ( mui_fds_first_token(ui) )
255 {
256 do
257 {
258 n++;
259 } while ( mui_fds_next_token(ui) );
260 }
261 return n;
262}

267/*
268 s must point to a valid command within FDS
269 return
270 The complete length of the command (including any text part)
271 sideeffect:
272 Any existing text part will be copied into ui->text
273 ui->text will be assigned to empty string if there is no text argument
274*/
275static size_t mui_fds_get_cmd_size(mui_t *ui, fds_t *s) MUI_NOINLINE;
276static size_t mui_fds_get_cmd_size(mui_t *ui, fds_t *s)
277{
278 size_t l = mui_fds_get_cmd_size_without_text(s);
279 uint8_t c = mui_get_fds_char(s);
280 ui->text[0] = '\0' ; /* always reset the text buffer */
281 if ( mui_fds_is_text(c) )
282 {
283 l += mui_fds_parse_text(ui, s+l);
284 }
285 return l;
286}

322/*
323 assumes a valid position in ui->fds and calculates all the other variables
324 some fields are always calculated like the ui->cmd and ui->len field
325 other member vars are calculated only if the return value is 1
326 will return 1 if the field id was found.
327 will return 0 if the field id was not found in uif or if ui->fds points to something else than a field
328*/
329static uint8_t mui_prepare_current_field(mui_t *ui) MUI_NOINLINE;
330static uint8_t mui_prepare_current_field(mui_t *ui)
331{
332 int muif_tidx;
333
334 ui->uif = NULL;
335 ui->dflags = 0;
336 ui->id0 = 0;
337 ui->id1 = 0;
338 ui->arg = 0;
339
340 /* calculate the length of the command and copy the text argument */
341 /* this will also clear the text in cases where there is no text argument */
342 ui->len = mui_fds_get_cmd_size(ui, ui->fds);
343 //printf("mui_prepare_current_field len=%d\n", ui->len);
344
345 /* get the command and check whether end of form is reached */
346 ui->cmd = mui_get_fds_char(ui->fds);
347 //printf("mui_prepare_current_field cmd='%c' len=%d\n", ui->cmd, ui->len);
348
349 /* Copy the cmd also to second id value. This is required for some commands, others will overwrite this below */
350 ui->id1 = ui->cmd;
351
352 /* now make the command uppercase so that both, upper and lower case are considered */
353 ui->cmd &= 0xdf; /* consider upper and lower case */
354
355 if ( ui->cmd == 'U' || ui->cmd == 0 )
356 return 0;
357
358 /* calculate the dynamic flags */
359 if ( ui->fds == ui->cursor_focus_fds )
360 ui->dflags |= MUIF_DFLAG_IS_CURSOR_FOCUS;
361 if ( ui->fds == ui->touch_focus_fds )
362 ui->dflags |= MUIF_DFLAG_IS_TOUCH_FOCUS;
363
364
365 /* get the id0 and id1 values */
366 if ( ui->cmd == 'F' || ui->cmd == 'B' || ui->cmd == 'T' || ui->cmd == 'A' )
367 {
368 ui->id0 = mui_get_fds_char(ui->fds+1);
369 ui->id1 = mui_get_fds_char(ui->fds+2);
370 ui->x = mui_get_fds_char(ui->fds+3);
371 ui->y = mui_get_fds_char(ui->fds+4);
372 if ( ui->cmd == 'A' || ui->cmd == 'T' )
373 {
374 ui->arg = mui_get_fds_char(ui->fds+5);
375 }
376 }
377 else if ( ui->cmd == 'D' || ui->cmd == 'Z' )
378 {
379 ui->id0 = mui_get_fds_char(ui->fds+1);
380 ui->id1 = mui_get_fds_char(ui->fds+2);
381 }
382 else if ( ui->cmd == 'S' )
383 {
384 ui->id0 = 'S';
385 ui->id1 = mui_get_fds_char(ui->fds+1);
386 }
387 else
388 {
389 ui->id0 = '.';
390 /* note that ui->id1 contains the original cmd value */
391 ui->x = mui_get_fds_char(ui->fds+1);
392 ui->y = mui_get_fds_char(ui->fds+2);
393 if ( ui->cmd == 'G' || ui->cmd == 'M' ) /* this is also true for 'g' or 'm' */
394 {
395 ui->arg = mui_get_fds_char(ui->fds+3);
396 }
397 }
398
399 //MUI_DEBUG("mui_prepare_current_field cmd='%c' len=%d arg=%d\n", ui->cmd, ui->len, ui->arg);
400
401
402 /* find the field */
403 muif_tidx = mui_find_uif(ui, ui->id0, ui->id1);
404 //printf("mui_prepare_current_field: muif_tidx=%d\n", muif_tidx);
405 if ( muif_tidx >= 0 )
406 {
407 ui->uif = ui->muif_tlist + muif_tidx;
408 return 1;
409 }
410 return 0;
411}

413/*
414 assumes that ui->fds has been assigned correctly
415 and that ui->target_fds and ui->tmp_fds had been cleared if required
416
417 Usually do not call this function directly, instead use mui_loop_over_form
418
419*/
420
421static void mui_inner_loop_over_form(mui_t *ui, uint8_t (*task)(mui_t *ui)) MUI_NOINLINE;
422static void mui_inner_loop_over_form(mui_t *ui, uint8_t (*task)(mui_t *ui))
423{
424 uint8_t cmd;
425
426 //MUI_DEBUG("mui_inner_loop_over_form start %p\n", task);
427
428 ui->fds += mui_fds_get_cmd_size(ui, ui->fds); // skip the first entry, it is U always
429 for(;;)
430 {
431 //printf("fds=%p *fds='%c'\n", ui->fds, ui->fds[0]);
432 /* get the command and check whether end of form is reached */
433 cmd = mui_get_fds_char(ui->fds);
434 if ( cmd == 'U' || cmd == 0 )
435 break;
436 if ( mui_prepare_current_field(ui) ) /* side effect: calculate ui->len */
437 if ( task(ui) ) /* call the task, which was provided as argument to this function */
438 {
439 //MUI_DEBUG("mui_inner_loop_over_form break by task\n");
440 break;
441 }
442 ui->fds += ui->len;
443 }
444
445 //MUI_DEBUG("mui_inner_loop_over_form end %p\n", task);
446}
447
448static void mui_loop_over_form(mui_t *ui, uint8_t (*task)(mui_t *ui)) MUI_NOINLINE;
449static void mui_loop_over_form(mui_t *ui, uint8_t (*task)(mui_t *ui))
450{
451 if ( mui_IsFormActive(ui) == 0 )
452 return;
453
454 ui->fds = ui->current_form_fds;
455 ui->target_fds = NULL;
456 ui->tmp_fds = NULL;
457
458 mui_inner_loop_over_form(ui, task);
459}

462 n is the form number
463*/
464static fds_t *mui_find_form(mui_t *ui, uint8_t n)
465{
466 fds_t *fds = ui->root_fds;
467 uint8_t cmd;
468
469 for( ;; )
470 {
471 cmd = mui_get_fds_char(fds);
472 if ( cmd == 0 )
473 break;
474 if ( cmd == 'U' )
475 {
476 if ( mui_get_fds_char(fds+1) == n )
477 {
478 return fds;
479 }
480 /* not found, just coninue */
481 }
482
483 fds += mui_fds_get_cmd_size(ui, fds);
484 }
485 return NULL;
486}

736 mui_EnterForm is similar to mui_GotoForm and differes only in the second argument (which is the form id instead of the fds pointer)

738void mui_EnterForm(mui_t *ui, fds_t *fds, uint8_t initial_cursor_position)
739{
740 /* exit any previous form, will not do anything if there is no current form */
741 mui_LeaveForm(ui);
742
743 /* clean focus fields */
744 ui->touch_focus_fds = NULL;
745 ui->cursor_focus_fds = NULL;
746
747 /* reset all the scoll values */
748 ui->form_scroll_top = 0;
749 ui->form_scroll_visible = 0;
750 ui->form_scroll_total = 0;
751
752 /* assign the form, which should be entered */
753 ui->current_form_fds = fds;
754
755 /* inform all fields that we start a new form */
756 MUI_DEBUG("mui_EnterForm: form_start, initial_cursor_position=%d\n", initial_cursor_position);
757 mui_loop_over_form(ui, mui_task_form_start);
758
759 /* assign initional cursor focus */
760 MUI_DEBUG("mui_EnterForm: find_first_cursor_uif\n");
761 mui_loop_over_form(ui, mui_task_find_first_cursor_uif);
762 ui->cursor_focus_fds = ui->target_fds; // NULL is ok
763 MUI_DEBUG("mui_EnterForm: find_first_cursor_uif target_fds=%p\n", ui->target_fds);
764
765 while( initial_cursor_position > 0 )
766 {
767 mui_NextField(ui); // mui_next_field(ui) is not sufficient in case of scrolling
768 initial_cursor_position--;
769 }
770
771 while( mui_send_cursor_enter_msg(ui) == 255 )
772 {
773 mui_NextField(ui); // mui_next_field(ui) is not sufficient in case of scrolling
774 }
775}
