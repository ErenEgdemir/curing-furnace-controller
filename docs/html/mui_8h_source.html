36 MUIF (Monochrome User Interface Functions)
37 n: A number 0 to 9 without any quotes, e.g.: 5
38 id: Exactly two characters or numbers in double quotes, e.g. "G5".
39 cb: A callback function with the following prototype: "uint8_t muif_cb(mui_t *ui, uint8_t msg)"
40 There are MANY predefined callback functions, see separate list
41 var: Address of a variable.
42 text: Normal text, but special characters might be required for some callback functions, for
43 example the text might contain a list of selectable elements separated with the '|' symbol.
44
45 MUIF_STYLE(n,cb)
46 Corresponding FDS command: MUI_STYLE(n)
47 Change the style of any other elements after MUI_STYLE(n), does not draw anything
48
49 MUIF_RO(id,cb)
50 Corresponding FDS command: MUI_DATA(id, text) MUI_XY(id, x, y), MUI_XYT(id, x,y,text), MUI_XYA(id, x,y,a), MUI_XYAT(id, x,y,a,text)
51 Places a read only element on the form.
52 The correct FDS command depends on the callback function.
53
54 MUIF_LABEL(cb)
55 Corresponding FDS command: MUI_LABEL(x,y,text)
56 Places a text at the specified position, similar to MUIF_RO
57
58 MUIF_GOTO(cb)
59 Corresponding FDS command: MUI_GOTO(x,y,n,text)
60 Places a button at the specified position, similar to MUIF_BUTTON, but does not require an ID.
61
62 MUIF_BUTTON(id,cb)
63 Corresponding FDS command: MUI_XY(id, x, y), MUI_XYT(id, x,y,text), MUI_XYA(id, x,y,a), MUI_XYAT(id, x,y,a,text)
64 Places a selectable element on the form.
65
66 MUIF_VARIABLE(id,var,cb)
67 Corresponding FDS command: MUI_XY(id, x, y), MUI_XYA(id, x,y,a)
68 Places a user input element at the specified location.
69 The correct FDS command depends on the callback function.
70
71
72*/

147struct muif_struct
148{
149 uint8_t id0;
150 uint8_t id1;
151 uint8_t cflags; // config flags e.g. MUIF_CFLAG_IS_CURSOR_SELECTABLE, if so, then it will not receive any cursor/touch msgs
152 uint8_t extra;
153 void *data; // might be a pointer to a variable
154 muif_cb cb; // callback
155} MUI_PROGMEM;

176#define MUIF_MSG_NONE 0
177#define MUIF_MSG_DRAW 1
178#define MUIF_MSG_FORM_START 2
179#define MUIF_MSG_FORM_END 3
180/* MUIF_MSG_CURSOR_ENTER return values: 255=skip this field, <255, continue*/
181#define MUIF_MSG_CURSOR_ENTER 4
182#define MUIF_MSG_CURSOR_SELECT 5
183
184/* optional VALUE messages, ignored by the mui core, but can be used inside the field functions */
185/* usually MUIF_MSG_VALUE_INCREMENT behaves like MUIF_MSG_CURSOR_SELECT */
186#define MUIF_MSG_VALUE_INCREMENT 6
187#define MUIF_MSG_VALUE_DECREMENT 7
188
189
190#define MUIF_MSG_CURSOR_LEAVE 8
191#define MUIF_MSG_TOUCH_DOWN 9
192#define MUIF_MSG_TOUCH_UP 10
193/* MUIF_MSG_EVENT_NEXT return values: 0=not handled, 1=handled, do nothing */
194/* If MUIF_MSG_EVENT_NEXT/PREV are NOT handled by the field function, then this msg will change the field */
195#define MUIF_MSG_EVENT_NEXT 11
196/* MUIF_MSG_EVENT_PREV return values: 0=not handled, 1=handled, do nothing */
197#define MUIF_MSG_EVENT_PREV 12
198
199/* dynamic flags */
200#define MUIF_DFLAG_IS_CURSOR_FOCUS 0x01
201#define MUIF_DFLAG_IS_TOUCH_FOCUS 0x02
202
203/* config flags */
204#define MUIF_CFLAG_IS_CURSOR_SELECTABLE 0x01
205#define MUIF_CFLAG_IS_TOUCH_SELECTABLE 0x02
206#define MUIF_CFLAG_IS_EXECUTE_ON_SELECT 0x04

209/* end user MUIF entries */
210#define MUIF(id,cflags,data,cb) { id[0], id[1], cflags, 0, data, cb}
211#define MUIF_STYLE(n,cb) MUIF("S" #n, 0, 0, cb)
212#define MUIF_RO(id,cb) MUIF(id,0, 0,cb)
213#define MUIF_LABEL(cb) MUIF(".L",0, 0,cb)
214#define MUIF_GOTO(cb) MUIF(".G",MUIF_CFLAG_IS_CURSOR_SELECTABLE,0,cb)
215#define MUIF_BUTTON(id,cb) MUIF(id,MUIF_CFLAG_IS_CURSOR_SELECTABLE,0,cb)
216#define MUIF_EXECUTE_ON_SELECT_BUTTON(id,cb) MUIF(id,MUIF_CFLAG_IS_CURSOR_SELECTABLE|MUIF_CFLAG_IS_EXECUTE_ON_SELECT,0,cb)
