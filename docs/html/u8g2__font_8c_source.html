42/*
43 font data:
44
45 offset bytes description
46 0 1 glyph_cnt number of glyphs
47 1 1 bbx_mode 0: proportional, 1: common height, 2: monospace, 3: multiple of 8
48 2 1 bits_per_0 glyph rle parameter
49 3 1 bits_per_1 glyph rle parameter
50
51 4 1 bits_per_char_width glyph rle parameter
52 5 1 bits_per_char_height glyph rle parameter
53 6 1 bits_per_char_x glyph rle parameter
54 7 1 bits_per_char_y glyph rle parameter
55 8 1 bits_per_delta_x glyph rle parameter
56
57 9 1 max_char_width
58 10 1 max_char_height
59 11 1 x offset
60 12 1 y offset (descent)
61
62 13 1 ascent (capital A)
63 14 1 descent (lower g)
64 15 1 ascent '('
65 16 1 descent ')'
66
67 17 1 start pos 'A' high byte
68 18 1 start pos 'A' low byte
69
70 19 1 start pos 'a' high byte
71 20 1 start pos 'a' low byte
72
73 21 1 start pos unicode high byte
74 22 1 start pos unicode low byte
75
76 Font build mode, 0: proportional, 1: common height, 2: monospace, 3: multiple of 8
77
78 Font build mode 0:
79 - "t"
80 - Ref height mode: U8G2_FONT_HEIGHT_MODE_TEXT, U8G2_FONT_HEIGHT_MODE_XTEXT or U8G2_FONT_HEIGHT_MODE_ALL
81 - use in transparent mode only (does not look good in solid mode)
82 - most compact format
83 - different font heights possible
84
85 Font build mode 1:
86 - "h"
87 - Ref height mode: U8G2_FONT_HEIGHT_MODE_ALL
88 - transparent or solid mode
89 - The height of the glyphs depend on the largest glyph in the font. This means font height depends on postfix "r", "f" and "n".
90

169/* calculate the overall length of the font, only used to create the picture for the google wiki */
170size_t u8g2_GetFontSize(const uint8_t *font_arg)
171{
172 uint16_t e;
173 const uint8_t *font = font_arg;
174 font += U8G2_FONT_DATA_STRUCT_SIZE;
175
176 for(;;)
177 {
178 if ( u8x8_pgm_read( font + 1 ) == 0 )
179 break;
180 font += u8x8_pgm_read( font + 1 );
181 }
182
183 /* continue with unicode section */
184 font += 2;
185
186 /* skip unicode lookup table */
187 font += u8g2_font_get_word(font, 0);
188
189 for(;;)
190 {
191 e = u8x8_pgm_read( font );
192 e <<= 8;
193 e |= u8x8_pgm_read( font + 1 );
194 if ( e == 0 )
195 break;
196 font += u8x8_pgm_read( font + 2 );
197 }
198
199 return (font - font_arg) + 2;
200}

237uint8_t u8g2_font_decode_get_unsigned_bits(u8g2_font_decode_t *f, uint8_t cnt)
238{
239 uint8_t val;
240 uint8_t bit_pos = f->decode_bit_pos;
241 uint8_t bit_pos_plus_cnt;
242
243 //val = *(f->decode_ptr);
244 val = u8x8_pgm_read( f->decode_ptr );
245
246 val >>= bit_pos;
247 bit_pos_plus_cnt = bit_pos;
248 bit_pos_plus_cnt += cnt;
249 if ( bit_pos_plus_cnt >= 8 )
250 {
251 uint8_t s = 8;
252 s -= bit_pos;
253 f->decode_ptr++;
254 //val |= *(f->decode_ptr) << (8-bit_pos);
255 val |= u8x8_pgm_read( f->decode_ptr ) << (s);
256 //bit_pos -= 8;
257 bit_pos_plus_cnt -= 8;
258 }
259 val &= (1U<<cnt)-1;
260 //bit_pos += cnt;
261
262 f->decode_bit_pos = bit_pos_plus_cnt;
263 return val;
264}

268 2 bit --> cnt = 2

271 3 bit --> cnt = 3
272 -2,-1,0. 1
273 -4,-3,-2,-1,0,1,2,3

276 r = bits(x-1)+1;
277 else
278 r = bits(x)+1;

283{
284 int8_t v, d;
285 v = (int8_t)u8g2_font_decode_get_unsigned_bits(f, cnt);
286 d = 1;
287 cnt--;
288 d <<= cnt;
289 v -= d;
290 return v;

369/*
370 Description:
371 Draw a run-length area of the glyph. "len" can have any size and the line
372 length has to be wrapped at the glyph border.
373 Args:
374 len: Length of the line
375 is_foreground foreground/background?
376 u8g2->font_decode.target_x X position
377 u8g2->font_decode.target_y Y position
378 u8g2->font_decode.is_transparent Transparent mode
379 Return:
380 -
381 Calls:
382 u8g2_Draw90Line()
383 Called by:
384 u8g2_font_decode_glyph()
385*/
386/* optimized */
387void u8g2_font_decode_len(u8g2_t *u8g2, uint8_t len, uint8_t is_foreground)
388{
389 uint8_t cnt; /* total number of remaining pixels, which have to be drawn */
390 uint8_t rem; /* remaining pixel to the right edge of the glyph */
391 uint8_t current; /* number of pixels, which need to be drawn for the draw procedure */
392 /* current is either equal to cnt or equal to rem */
393
394 /* local coordinates of the glyph */
395 uint8_t lx,ly;
396
397 /* target position on the screen */
398 u8g2_uint_t x, y;
399
400 u8g2_font_decode_t *decode = &(u8g2->font_decode);
401
402 cnt = len;
403
404 /* get the local position */
405 lx = decode->x;
406 ly = decode->y;
407
408 for(;;)
409 {
410 /* calculate the number of pixel to the right edge of the glyph */
411 rem = decode->glyph_width;
412 rem -= lx;
413
414 /* calculate how many pixel to draw. This is either to the right edge */
415 /* or lesser, if not enough pixel are left */
416 current = rem;
417 if ( cnt < rem )
418 current = cnt;
419
420
421 /* now draw the line, but apply the rotation around the glyph target position */
422 //u8g2_font_decode_draw_pixel(u8g2, lx,ly,current, is_foreground);
423
424 /* get target position */
425 x = decode->target_x;
426 y = decode->target_y;
427
428 /* apply rotation */
429#ifdef U8G2_WITH_FONT_ROTATION
430
431 x = u8g2_add_vector_x(x, lx, ly, decode->dir);
432 y = u8g2_add_vector_y(y, lx, ly, decode->dir);
433
434 //u8g2_add_vector(&x, &y, lx, ly, decode->dir);
435
436#else
437 x += lx;
438 y += ly;
439#endif
440
441 /* draw foreground and background (if required) */
442 if ( is_foreground )
443 {
444 u8g2->draw_color = decode->fg_color; /* draw_color will be restored later */
445 u8g2_DrawHVLine(u8g2,
446 x,
447 y,
448 current,
449#ifdef U8G2_WITH_FONT_ROTATION
450 /* dir */ decode->dir
451#else
452 0
453#endif
454 );
455 }
456 else if ( decode->is_transparent == 0 )
457 {
458 u8g2->draw_color = decode->bg_color; /* draw_color will be restored later */
459 u8g2_DrawHVLine(u8g2,
460 x,
461 y,
462 current,
463#ifdef U8G2_WITH_FONT_ROTATION
464 /* dir */ decode->dir
465#else
466 0
467#endif
468 );
469 }
470
471 /* check, whether the end of the run length code has been reached */
472 if ( cnt < rem )
473 break;
474 cnt -= rem;
475 lx = 0;
476 ly++;
477 }
478 lx += cnt;
479
480 decode->x = lx;
481 decode->y = ly;
482}

598/*
599 Description:
600 Decode and draw a glyph.
601 Args:
602 glyph_data: Pointer to the compressed glyph data of the font
603 u8g2->font_decode.target_x X position
604 u8g2->font_decode.target_y Y position
605 u8g2->font_decode.is_transparent Transparent mode
606 Return:
607 Width (delta x advance) of the glyph.
608 Calls:
609 u8g2_font_decode_len()
610*/
611/* optimized */
612int8_t u8g2_font_decode_glyph(u8g2_t *u8g2, const uint8_t *glyph_data)
613{
614 uint8_t a, b;
615 int8_t x, y;
616 int8_t d;
617 int8_t h;
618 u8g2_font_decode_t *decode = &(u8g2->font_decode);
619
620 u8g2_font_setup_decode(u8g2, glyph_data); /* set values in u8g2->font_decode data structure */
621 h = u8g2->font_decode.glyph_height;
622
623 x = u8g2_font_decode_get_signed_bits(decode, u8g2->font_info.bits_per_char_x);
624 y = u8g2_font_decode_get_signed_bits(decode, u8g2->font_info.bits_per_char_y);
625 d = u8g2_font_decode_get_signed_bits(decode, u8g2->font_info.bits_per_delta_x);
626
627 if ( decode->glyph_width > 0 )
628 {
629#ifdef U8G2_WITH_FONT_ROTATION
630 decode->target_x = u8g2_add_vector_x(decode->target_x, x, -(h+y), decode->dir);
631 decode->target_y = u8g2_add_vector_y(decode->target_y, x, -(h+y), decode->dir);
632
633 //u8g2_add_vector(&(decode->target_x), &(decode->target_y), x, -(h+y), decode->dir);
634
635#else
636 decode->target_x += x;
637 decode->target_y -= h+y;
638#endif
639 //u8g2_add_vector(&(decode->target_x), &(decode->target_y), x, -(h+y), decode->dir);
640
641#ifdef U8G2_WITH_INTERSECTION
642 {
643 u8g2_uint_t x0, x1, y0, y1;
644 x0 = decode->target_x;
645 y0 = decode->target_y;
646 x1 = x0;
647 y1 = y0;
648
649#ifdef U8G2_WITH_FONT_ROTATION
650 switch(decode->dir)
651 {
652 case 0:
653 x1 += decode->glyph_width;
654 y1 += h;
655 break;
656 case 1:
657 x0 -= h;
658 x0++; /* shift down, because of assymetric boundaries for the interseciton test */
659 x1++;
660 y1 += decode->glyph_width;
661 break;
662 case 2:
663 x0 -= decode->glyph_width;
664 x0++; /* shift down, because of assymetric boundaries for the interseciton test */
665 x1++;
666 y0 -= h;
667 y0++; /* shift down, because of assymetric boundaries for the interseciton test */
668 y1++;
669 break;
670 case 3:
671 x1 += h;
672 y0 -= decode->glyph_width;
673 y0++; /* shift down, because of assymetric boundaries for the interseciton test */
674 y1++;
675 break;
676 }
677#else /* U8G2_WITH_FONT_ROTATION */
678 x1 += decode->glyph_width;
679 y1 += h;
680#endif
681
682 if ( u8g2_IsIntersection(u8g2, x0, y0, x1, y1) == 0 )
683 return d;
684 }
685#endif /* U8G2_WITH_INTERSECTION */
686
687 /* reset local x/y position */
688 decode->x = 0;
689 decode->y = 0;
690
691 /* decode glyph */
692 for(;;)
693 {
694 a = u8g2_font_decode_get_unsigned_bits(decode, u8g2->font_info.bits_per_0);
695 b = u8g2_font_decode_get_unsigned_bits(decode, u8g2->font_info.bits_per_1);
696 do
697 {
698 u8g2_font_decode_len(u8g2, a, 0);
699 u8g2_font_decode_len(u8g2, b, 1);
700 } while( u8g2_font_decode_get_unsigned_bits(decode, 1) != 0 );
701
702 if ( decode->y >= h )
703 break;
704 }

713int8_t u8g2_font_2x_decode_glyph(u8g2_t *u8g2, const uint8_t *glyph_data)
714{
715 uint8_t a, b;
716 int8_t x, y;
717 int8_t d;
718 int8_t h;
719 u8g2_font_decode_t *decode = &(u8g2->font_decode);
720
721 u8g2_font_setup_decode(u8g2, glyph_data); /* set values in u8g2->font_decode data structure */
722 h = u8g2->font_decode.glyph_height;
723
724 x = u8g2_font_decode_get_signed_bits(decode, u8g2->font_info.bits_per_char_x);
725 y = u8g2_font_decode_get_signed_bits(decode, u8g2->font_info.bits_per_char_y);
726 d = u8g2_font_decode_get_signed_bits(decode, u8g2->font_info.bits_per_delta_x);
727
728 if ( decode->glyph_width > 0 )
729 {
730 decode->target_x += x;
731 decode->target_y -= 2*h+y;
732
733#ifdef U8G2_WITH_INTERSECTION
734 {
735 u8g2_uint_t x0, x1, y0, y1;
736 x0 = decode->target_x;
737 y0 = decode->target_y;
738 x1 = x0;
739 y1 = y0;
740
741 x1 += 2*decode->glyph_width;
742 y1 += 2*h;
743
744 if ( u8g2_IsIntersection(u8g2, x0, y0, x1, y1) == 0 )
745 return 2*d;
746 }
747#endif /* U8G2_WITH_INTERSECTION */
748
749 /* reset local x/y position */
750 decode->x = 0;
751 decode->y = 0;
752
753 /* decode glyph */
754 for(;;)
755 {
756 a = u8g2_font_decode_get_unsigned_bits(decode, u8g2->font_info.bits_per_0);
757 b = u8g2_font_decode_get_unsigned_bits(decode, u8g2->font_info.bits_per_1);
758 do
759 {
760 u8g2_font_2x_decode_len(u8g2, a, 0);
761 u8g2_font_2x_decode_len(u8g2, b, 1);
762 } while( u8g2_font_decode_get_unsigned_bits(decode, 1) != 0 );
763
764 if ( decode->y >= h )
765 break;
766 }
767
768 /* restore the u8g2 draw color, because this is modified by the decode algo */
769 u8g2->draw_color = decode->fg_color;
770 }
771 return d*2;
772}

797 }

799 for(;;)
800 {
801 if ( u8x8_pgm_read( font + 1 ) == 0 )
802 break;

804 {
805 return font+2; /* skip encoding and glyph size */
806 }

808 }
809 }
810#ifdef U8G2_WITH_UNICODE
811 else
812 {
813 uint16_t e;
814 const uint8_t *unicode_lookup_table;
815
816// removed, there is now the new index table
817//#ifdef __unix__

1566
1567 The data argument should look like this:
1568 static const unsigned char teststring[] U8X8_PROGMEM = {
1569 0x09, 0x28, 0x00, 0x00, // u8g2_DrawGlyph(&u8g2, 0, 0, 2344);

1577

1579 <encoding high byte> <encoding low byte> <delta-x> <delta-y>
1580 The last row is marked with encoding=0
1581
1582 The algorithm is
1583 Input: x,y
1584 with all rows (until encoding is 0)
1585 x += <delta-x>
1586 y += <delta-y>
1587 draw glyph with <encoding> at x, y
1588
1589 A call to this function will require transparent mode:
1590 u8g2_SetFontMode(&u8g2, 1);
1591
1592 Limitation:
1593 Glyph delta must be lower than 128, this bascially means, that the glyph size is limited to
1594 hight/width of 128 pixel
1595
1596 Further details:
1597 https://github.com/olikraus/u8g2/issues/2656
1598
1599*/
1600void u8g2_DrawHB(u8g2_t *u8g2, u8g2_uint_t x, u8g2_uint_t y, const unsigned char *data)
1601{
1602 uint16_t encoding = 0;
1603 for (;;)
1604 {
1605 encoding = u8x8_pgm_read(data);
1606 data++;
1607 encoding <<= 8;
1608 encoding |= u8x8_pgm_read(data);
1609 data++;
1610 if (encoding == 0)
1611 break;
1612 x += (int8_t)u8x8_pgm_read(data);
1613 data++;
1614 y += (int8_t)u8x8_pgm_read(data);
1615 data++;

1617 }
