41static void u8g2_draw_arc(u8g2_t *u8g2, u8g2_uint_t x0, u8g2_uint_t y0, u8g2_uint_t rad, uint8_t start, uint8_t end)
42{
43 // Manage angle inputs
44 uint8_t full = (start == end);
45 uint8_t inverted = (start > end);
46 uint8_t a_start = inverted ? end : start;
47 uint8_t a_end = inverted ? start : end;
48
49 // Initialize variables
50 uint32_t ratio;
51 u8g2_int_t x = 0;
52 u8g2_int_t y = rad;
53 u8g2_int_t d = rad - 1;
54
55 // Trace arc radius with the Andres circle algorithm (process each pixel of a 1/8th circle of radius rad)
56 while (y >= x)
57 {
58 // Get the percentage of 1/8th circle drawn with a fast approximation of arctan(x/y)
59 ratio = x * 255 / y; // x/y [0..255]
60 ratio = ratio * (770195 - (ratio - 255) * (ratio + 941)) / 6137491; // arctan(x/y) [0..32]
61
62 // Fill the pixels of the 8 sections of the circle, but only on the arc defined by the angles (start and end)
63 if(full || ((ratio >= a_start && ratio < a_end) ^ inverted)) u8g2_DrawPixel(u8g2, x0 + y, y0 - x);
64 if(full || (((ratio + a_end) > 63 && (ratio + a_start) <= 63) ^ inverted)) u8g2_DrawPixel(u8g2, x0 + x, y0 - y);
65 if(full || (((ratio + 64) >= a_start && (ratio + 64) < a_end) ^ inverted)) u8g2_DrawPixel(u8g2, x0 - x, y0 - y);
66 if(full || (((ratio + a_end) > 127 && (ratio + a_start) <= 127) ^ inverted)) u8g2_DrawPixel(u8g2, x0 - y, y0 - x);
67 if(full || (((ratio + 128) >= a_start && (ratio + 128) < a_end) ^ inverted)) u8g2_DrawPixel(u8g2, x0 - y, y0 + x);
68 if(full || (((ratio + a_end) > 191 && (ratio + a_start) <= 191) ^ inverted)) u8g2_DrawPixel(u8g2, x0 - x, y0 + y);
69 if(full || (((ratio + 192) >= a_start && (ratio + 192) < a_end) ^ inverted)) u8g2_DrawPixel(u8g2, x0 + x, y0 + y);
70 if(full || (((ratio + a_end) > 255 && (ratio + a_start) <= 255) ^ inverted)) u8g2_DrawPixel(u8g2, x0 + y, y0 + x);
71
72 // Run Andres circle algorithm to get to the next pixel
73 if (d >= 2 * x)
74 {
75 d = d - 2 * x - 1;
76 x = x + 1;
77 }
78 else if (d < 2 * (rad - y))
79 {
80 d = d + 2 * y - 1;
81 y = y - 1;
82 }
83 else
84 {
85 d = d + 2 * (y - x - 1);
86 y = y - 1;
87 x = x + 1;
88 }
89 }
90}
