43/*
44 Draw a string at x,y
45 Center string within w (left adjust if w < pixel len of s)
46
47 Side effects:
48 u8g2_SetFontDirection(u8g2, 0);
49 u8g2_SetFontPosBaseline(u8g2);
50
51*/
52void u8g2_DrawUTF8Line(u8g2_t *u8g2, u8g2_uint_t x, u8g2_uint_t y, u8g2_uint_t w, const char *s, uint8_t border_size, uint8_t is_invert)
53{
54 u8g2_uint_t d, str_width;
55 u8g2_uint_t fx, fy, fw, fh;
56
57 /* only horizontal strings are supported, so force this here */
58 u8g2_SetFontDirection(u8g2, 0);
59
60 /* revert y position back to baseline ref */
61 y += u8g2->font_calc_vref(u8g2);
62
63 /* calculate the width of the string in pixel */
64 str_width = u8g2_GetUTF8Width(u8g2, s);
65
66#ifdef U8G2_BALANCED_STR_WIDTH_CALCULATION
67 /* subtract the first character offset added by the width calculation */
68 str_width -= u8g2_GetXOffsetUTF8(u8g2, s);
69#endif
70
71 /* calculate delta d within the box */
72 d = 0;
73 if ( str_width < w )
74 {
75 d = w;
76 d -=str_width;
77 d /= 2;
78 }
79 else
80 {
81 w = str_width;
82 }
83
84 /* caluclate text box */
85 fx = x;
86 fy = y - u8g2_GetAscent(u8g2) ;
87 fw = w;
88 fh = u8g2_GetAscent(u8g2) - u8g2_GetDescent(u8g2) ;
89
90 /* draw the box, if inverted */
91 u8g2_SetDrawColor(u8g2, 1);
92 if ( is_invert )
93 {
94 u8g2_DrawBox(u8g2, fx, fy, fw, fh);
95 }
96
97 /* draw the frame */
98 while( border_size > 0 )
99 {
100 fx--;
101 fy--;
102 fw +=2;
103 fh +=2;
104 u8g2_DrawFrame(u8g2, fx, fy, fw, fh );
105 border_size--;
106 }
107
108 if ( is_invert )
109 {
110 u8g2_SetDrawColor(u8g2, 0);
111 }
112 else
113 {
114 u8g2_SetDrawColor(u8g2, 1);
115 }
116
117 /* draw the text */
118 u8g2_DrawUTF8(u8g2, x+d, y, s);
119
120 /* revert draw color */
121 u8g2_SetDrawColor(u8g2, 1);
122
123}

126/*
127 draw several lines at position x,y.
128 lines are stored in s and must be separated with '\n'.
129 lines can be centered with respect to "w"
130 if s == NULL nothing is drawn and 0 is returned
131 returns the number of lines in s multiplied with line_height
132*/
133u8g2_uint_t u8g2_DrawUTF8Lines(u8g2_t *u8g2, u8g2_uint_t x, u8g2_uint_t y, u8g2_uint_t w, u8g2_uint_t line_height, const char *s)
134{
135 uint8_t i;
136 uint8_t cnt;
137 u8g2_uint_t yy = 0;
138 cnt = u8x8_GetStringLineCnt(s);
139 //printf("str=%s\n", s);
140 //printf("cnt=%d, y=%d, line_height=%d\n", cnt, y, line_height);
141 for( i = 0; i < cnt; i++ )
142 {
143 //printf(" i=%d, y=%d, line_height=%d\n", i, y, line_height);
144 u8g2_DrawUTF8Line(u8g2, x, y, w, u8x8_GetStringLineStart(i, s), 0, 0);
145 y+=line_height;
146 yy+=line_height;
147 }
148 return yy;
149}

197/*
198 title: NULL for no title, valid str for title line. Can contain mutliple lines, separated by '\n'
199 start_pos: default position for the cursor, first line is 1.
200 sl: string list (list of strings separated by \n)
201 returns 0 if user has pressed the home key
202 returns the selected line if user has pressed the select key
203 side effects:
204 u8g2_SetFontDirection(u8g2, 0);
205 u8g2_SetFontPosBaseline(u8g2);
206
207*/
208uint8_t u8g2_UserInterfaceSelectionList(u8g2_t *u8g2, const char *title, uint8_t start_pos, const char *sl)
209{
210 u8sl_t u8sl;
211 u8g2_uint_t yy;
212
213 uint8_t event;
214
215 u8g2_uint_t line_height = u8g2_GetAscent(u8g2) - u8g2_GetDescent(u8g2)+MY_BORDER_SIZE;
216
217 uint8_t title_lines = u8x8_GetStringLineCnt(title);
218 uint8_t display_lines;
219
220
221 if ( start_pos > 0 ) /* issue 112 */
222 start_pos--; /* issue 112 */
223
224
225 if ( title_lines > 0 )
226 {
227 display_lines = (u8g2_GetDisplayHeight(u8g2)-3) / line_height;
228 u8sl.visible = display_lines;
229 u8sl.visible -= title_lines;
230 }
231 else
232 {
233 display_lines = u8g2_GetDisplayHeight(u8g2) / line_height;
234 u8sl.visible = display_lines;
235 }
236
237 u8sl.total = u8x8_GetStringLineCnt(sl);
238 u8sl.first_pos = 0;
239 u8sl.current_pos = start_pos;
240
241 if ( u8sl.current_pos >= u8sl.total )
242 u8sl.current_pos = u8sl.total-1;
243 if ( u8sl.first_pos+u8sl.visible <= u8sl.current_pos )
244 u8sl.first_pos = u8sl.current_pos-u8sl.visible+1;
245
246 u8g2_SetFontPosBaseline(u8g2);
247
248 for(;;)
249 {
250 u8g2_FirstPage(u8g2);
251 do
252 {
253 yy = u8g2_GetAscent(u8g2);
254 if ( title_lines > 0 )
255 {
256 yy += u8g2_DrawUTF8Lines(u8g2, 0, yy, u8g2_GetDisplayWidth(u8g2), line_height, title);
257
258 u8g2_DrawHLine(u8g2, 0, yy-line_height- u8g2_GetDescent(u8g2) + 1, u8g2_GetDisplayWidth(u8g2));
259
260 yy += 3;
261 }
262 u8g2_DrawSelectionList(u8g2, &u8sl, yy, sl);
263 } while( u8g2_NextPage(u8g2) );
264
265#ifdef U8G2_REF_MAN_PIC
266 return 0;
267#endif
268
269
270 for(;;)
271 {
272 event = u8x8_GetMenuEvent(u8g2_GetU8x8(u8g2));
273 if ( event == U8X8_MSG_GPIO_MENU_SELECT )
274 return u8sl.current_pos+1; /* +1, issue 112 */
275 else if ( event == U8X8_MSG_GPIO_MENU_HOME )
276 return 0; /* issue 112: return 0 instead of start_pos */
277 else if ( event == U8X8_MSG_GPIO_MENU_NEXT || event == U8X8_MSG_GPIO_MENU_DOWN )
278 {
279 u8sl_Next(&u8sl);
280 break;
281 }
282 else if ( event == U8X8_MSG_GPIO_MENU_PREV || event == U8X8_MSG_GPIO_MENU_UP )
283 {
284 u8sl_Prev(&u8sl);
285 break;
286 }
287 }
288 }
