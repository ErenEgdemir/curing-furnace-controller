130/*
131 update dimension:
132 calculate the following variables:
133 u8g2_uint_t buf_x0; left corner of the buffer
134 u8g2_uint_t buf_x1; right corner of the buffer (excluded)
135 u8g2_uint_t buf_y0;
136 u8g2_uint_t buf_y1;
137*/
138
139static void u8g2_update_dimension_common(u8g2_t *u8g2)
140{
141 const u8x8_display_info_t *display_info = u8g2_GetU8x8(u8g2)->display_info;
142 u8g2_uint_t t;
143
144 t = u8g2->tile_buf_height;
145 t *= 8;
146 u8g2->pixel_buf_height = t;
147
148 t = display_info->tile_width;
149#ifndef U8G2_16BIT
150 if ( t >= 32 )
151 t = 31;
152#endif
153 t *= 8;
154 u8g2->pixel_buf_width = t;
155
156 t = u8g2->tile_curr_row;
157 t *= 8;
158 u8g2->pixel_curr_row = t;
159
160 t = u8g2->tile_buf_height;
161 /* handle the case, where the buffer is larger than the (remaining) part of the display */
162 if ( t + u8g2->tile_curr_row > display_info->tile_height )
163 t = display_info->tile_height - u8g2->tile_curr_row;
164 t *= 8;
165
166 u8g2->buf_y0 = u8g2->pixel_curr_row;
167 u8g2->buf_y1 = u8g2->buf_y0;
168 u8g2->buf_y1 += t;
169
170
171#ifdef U8G2_16BIT
172 u8g2->width = display_info->pixel_width;
173 u8g2->height = display_info->pixel_height;
174#else
175 u8g2->width = 240;
176 if ( display_info->pixel_width <= 240 )
177 u8g2->width = display_info->pixel_width;
178 u8g2->height = display_info->pixel_height;
179#endif
180
181}

186#ifdef U8G2_WITH_CLIP_WINDOW_SUPPORT
187static void u8g2_apply_clip_window(u8g2_t *u8g2)
188{
189 /* check aganst the current user_??? window */
190 if ( u8g2_IsIntersection(u8g2, u8g2->clip_x0, u8g2->clip_y0, u8g2->clip_x1, u8g2->clip_y1) == 0 )
191 {
192 u8g2->is_page_clip_window_intersection = 0;
193 }
194 else
195 {
196 u8g2->is_page_clip_window_intersection = 1;
197
198 if ( u8g2->user_x0 < u8g2->clip_x0 )
199 u8g2->user_x0 = u8g2->clip_x0;
200 if ( u8g2->user_x1 > u8g2->clip_x1 )
201 u8g2->user_x1 = u8g2->clip_x1;
202 if ( u8g2->user_y0 < u8g2->clip_y0 )
203 u8g2->user_y0 = u8g2->clip_y0;
204 if ( u8g2->user_y1 > u8g2->clip_y1 )
205 u8g2->user_y1 = u8g2->clip_y1;
206 }
207}
208#endif /* U8G2_WITH_CLIP_WINDOW_SUPPORT */
