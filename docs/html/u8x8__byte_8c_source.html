107uint8_t u8x8_byte_4wire_sw_spi(u8x8_t *u8x8, uint8_t msg, uint8_t arg_int, void *arg_ptr)
108{
109 uint8_t i, b;
110 uint8_t *data;
111 uint8_t takeover_edge = u8x8_GetSPIClockPhase(u8x8);
112 uint8_t not_takeover_edge = 1 - takeover_edge;
113
114 switch(msg)
115 {
116 case U8X8_MSG_BYTE_SEND:
117 data = (uint8_t *)arg_ptr;
118 while( arg_int > 0 )
119 {
120 b = *data;
121 data++;
122 arg_int--;
123 for( i = 0; i < 8; i++ )
124 {
125 if ( b & 128 )
126 u8x8_gpio_SetSPIData(u8x8, 1);
127 else
128 u8x8_gpio_SetSPIData(u8x8, 0);
129 b <<= 1;
130
131 u8x8_gpio_SetSPIClock(u8x8, not_takeover_edge);
132 u8x8_gpio_Delay(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->sda_setup_time_ns);
133 u8x8_gpio_SetSPIClock(u8x8, takeover_edge);
134 u8x8_gpio_Delay(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->sck_pulse_width_ns);
135 }
136 }
137 break;
138
139 case U8X8_MSG_BYTE_INIT:
140 /* disable chipselect */
141 u8x8_gpio_SetCS(u8x8, u8x8->display_info->chip_disable_level);
142 /* no wait required here */
143
144 /* for SPI: setup correct level of the clock signal */
145 u8x8_gpio_SetSPIClock(u8x8, u8x8_GetSPIClockPhase(u8x8));
146 break;
147 case U8X8_MSG_BYTE_SET_DC:
148 u8x8_gpio_SetDC(u8x8, arg_int);
149 break;
150 case U8X8_MSG_BYTE_START_TRANSFER:
151 u8x8_gpio_SetCS(u8x8, u8x8->display_info->chip_enable_level);
152 u8x8->gpio_and_delay_cb(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->post_chip_enable_wait_ns, NULL);
153 break;
154 case U8X8_MSG_BYTE_END_TRANSFER:
155 u8x8->gpio_and_delay_cb(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->pre_chip_disable_wait_ns, NULL);
156 u8x8_gpio_SetCS(u8x8, u8x8->display_info->chip_disable_level);
157 break;
158 default:
159 return 0;
160 }
161 return 1;
162}

175 while( arg_int > 0 )
176 {
177 b = *data;
178 data++;
179 arg_int--;
180 for( i = U8X8_MSG_GPIO_D0; i <= U8X8_MSG_GPIO_D7; i++ )

196 /* ensure that the enable signal is high */

209 break;
210 default:
211 return 0;
212 }

268uint8_t u8x8_byte_3wire_sw_spi(u8x8_t *u8x8, uint8_t msg, uint8_t arg_int, void *arg_ptr)
269{
270 uint8_t i;
271 uint8_t *data;
272 uint8_t takeover_edge = u8x8_GetSPIClockPhase(u8x8);
273 uint8_t not_takeover_edge = 1 - takeover_edge;
274 uint16_t b;
275 static uint8_t last_dc;
276
277 switch(msg)
278 {
279 case U8X8_MSG_BYTE_SEND:
280 data = (uint8_t *)arg_ptr;
281 while( arg_int > 0 )
282 {
283 b = *data;
284 if ( last_dc != 0 )
285 b |= 256;
286 data++;
287 arg_int--;
288 for( i = 0; i < 9; i++ )
289 {
290 if ( b & 256 )
291 u8x8_gpio_SetSPIData(u8x8, 1);
292 else
293 u8x8_gpio_SetSPIData(u8x8, 0);
294 b <<= 1;
295
296 u8x8_gpio_SetSPIClock(u8x8, not_takeover_edge);
297 u8x8_gpio_Delay(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->sda_setup_time_ns);
298 u8x8_gpio_SetSPIClock(u8x8, takeover_edge);
299 u8x8_gpio_Delay(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->sck_pulse_width_ns);
300 }
301 }
302 break;
303
304 case U8X8_MSG_BYTE_INIT:
305 /* disable chipselect */
306 u8x8_gpio_SetCS(u8x8, u8x8->display_info->chip_disable_level);
307 /* no wait required here */
308
309 /* for SPI: setup correct level of the clock signal */
310 u8x8_gpio_SetSPIClock(u8x8, u8x8_GetSPIClockPhase(u8x8));
311 break;
312 case U8X8_MSG_BYTE_SET_DC:
313 last_dc = arg_int;
314 break;
315 case U8X8_MSG_BYTE_START_TRANSFER:
316 u8x8_gpio_SetCS(u8x8, u8x8->display_info->chip_enable_level);
317 u8x8->gpio_and_delay_cb(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->post_chip_enable_wait_ns, NULL);
318 break;
319 case U8X8_MSG_BYTE_END_TRANSFER:
320 u8x8->gpio_and_delay_cb(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->pre_chip_disable_wait_ns, NULL);
321 u8x8_gpio_SetCS(u8x8, u8x8->display_info->chip_disable_level);
322 break;
323 default:
324 return 0;
325 }
326 return 1;
327}

332{
333 u8x8_gpio_SetCS(u8x8, arg&1);

393/* sed1520 or sbn1661
394 U8X8_MSG_GPIO_E --> E1
395 U8X8_MSG_GPIO_CS --> E2
396*/
397uint8_t u8x8_byte_sed1520(u8x8_t *u8x8, uint8_t msg, uint8_t arg_int, void *arg_ptr)
398{
399 uint8_t i, b;
400 uint8_t *data;
401 static uint8_t enable_pin;
402
403 switch(msg)
404 {
405 case U8X8_MSG_BYTE_SEND:
406 data = (uint8_t *)arg_ptr;
407 while( arg_int > 0 )
408 {
409 b = *data;
410 data++;
411 arg_int--;
412 for( i = U8X8_MSG_GPIO_D0; i <= U8X8_MSG_GPIO_D7; i++ )
413 {
414 u8x8_gpio_call(u8x8, i, b&1);
415 b >>= 1;
416 }
417
418 u8x8_gpio_Delay(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->data_setup_time_ns);
419 u8x8_gpio_call(u8x8, enable_pin, 1);
420 u8x8_gpio_Delay(u8x8, U8X8_MSG_DELAY_NANO, 200); /* KS0108 requires 450 ns, use 200 here */
421 u8x8_gpio_Delay(u8x8, U8X8_MSG_DELAY_NANO, u8x8->display_info->write_pulse_width_ns); /* expect 250 here */
422 u8x8_gpio_call(u8x8, enable_pin, 0);
423 }
424 break;
425
426 case U8X8_MSG_BYTE_INIT:
427 /* disable chipselect */
428 u8x8_gpio_SetCS(u8x8, u8x8->display_info->chip_disable_level);
429 /* ensure that the enable signals are low */
430 u8x8_gpio_call(u8x8, U8X8_MSG_GPIO_E, 0);
431 u8x8_gpio_call(u8x8, U8X8_MSG_GPIO_CS, 0);
432 enable_pin = U8X8_MSG_GPIO_E;
433 break;
434 case U8X8_MSG_BYTE_SET_DC:
435 u8x8_gpio_SetDC(u8x8, arg_int);
436 break;
437 case U8X8_MSG_BYTE_START_TRANSFER:
438 /* cs lines are not supported for the SED1520/SBN1661 */
439 /* instead, this will select the E1 or E2 line */
440 /* arg_int is set by u8x8_d_sbn1661_122x32() function */
441 enable_pin = U8X8_MSG_GPIO_E;
442 if ( arg_int != 0 )
443 enable_pin = U8X8_MSG_GPIO_CS;
444 break;
445 case U8X8_MSG_BYTE_END_TRANSFER:
446 break;
447 default:
448 return 0;
449 }
450 return 1;
451}

503static void i2c_start(u8x8_t *u8x8)
504{
505 if ( u8x8->i2c_started != 0 )
506 {
507 /* if already started: do restart */
508 i2c_read_sda(u8x8); /* SDA = 1 */
509 i2c_delay(u8x8);
510 i2c_read_scl_and_delay(u8x8);
511 }
512 i2c_read_sda(u8x8);
513 /* send the start condition, both lines go from 1 to 0 */
514 i2c_clear_sda(u8x8);
515 i2c_delay(u8x8);
516 i2c_clear_scl(u8x8);
517 u8x8->i2c_started = 1;
518}
519
520
521static void i2c_stop(u8x8_t *u8x8)
522{
523 /* set SDA to 0 */
524 i2c_clear_sda(u8x8);
525 i2c_delay(u8x8);
526
527 /* now release all lines */
528 i2c_read_scl_and_delay(u8x8);
529
530 /* set SDA to 1 */
531 i2c_read_sda(u8x8);
532 i2c_delay(u8x8);
533 u8x8->i2c_started = 0;
534}
