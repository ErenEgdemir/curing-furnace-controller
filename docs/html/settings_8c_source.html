14static uint32_t crc32_calc(const void *data, size_t len)
15{
16 uint32_t crc = 0xFFFFFFFFu;
17 const uint8_t *p = (const uint8_t*)data;
18 while (len--) {
19 crc ^= *p++;
20 for (int i = 0; i < 8; i++) {
21 crc = (crc >> 1) ^ (0xEDB88320u & (-(int32_t)(crc & 1)));
22 }
23 }
24 return ~crc;
25}

32/* Adres→Sektör: 256 KB cihazda 0..5 mevcuttur; 512 KB’da 6 ve 7 de olabilir.
33 Sembol mevcut değilse (#ifdef) o dala hiç girmeyeceğiz. */
34static uint32_t GetSector(uint32_t Address)
35{
36 if (Address < 0x08004000u) return FLASH_SECTOR_0; // 16KB
37 else if (Address < 0x08008000u) return FLASH_SECTOR_1; // 16KB
38 else if (Address < 0x0800C000u) return FLASH_SECTOR_2; // 16KB
39 else if (Address < 0x08010000u) return FLASH_SECTOR_3; // 16KB
40 else if (Address < 0x08020000u) return FLASH_SECTOR_4; // 64KB
41 else if (Address < 0x08040000u) return FLASH_SECTOR_5; // 128KB
42 #ifdef FLASH_SECTOR_6
43 else if (Address < 0x08060000u) return FLASH_SECTOR_6; // 128KB
44 #endif
45 #ifdef FLASH_SECTOR_7
46 else return FLASH_SECTOR_7; // 128KB
47 #else
48 else return FLASH_SECTOR_5; // 256KB cihazda emniyetli dönüş
49 #endif
50}

55static int Flash_EraseRange(uint32_t start_addr, uint32_t end_addr_exclusive)
56{
57 FLASH_EraseInitTypeDef ei = {0};
58 uint32_t sector_error = 0;
59
60 uint32_t start_sector = GetSector(start_addr);
61 uint32_t end_sector = GetSector(end_addr_exclusive);
62
63 ei.TypeErase = FLASH_TYPEERASE_SECTORS;
64 ei.VoltageRange = FLASH_VR;
65 ei.Sector = start_sector;
66 ei.NbSectors = (end_sector - start_sector) + 1;
67
68 HAL_FLASH_Unlock();
69 HAL_StatusTypeDef st = HAL_FLASHEx_Erase(&ei, &sector_error);
70 HAL_FLASH_Lock();
71
72 return (st == HAL_OK) ? SETTINGS_OK : SETTINGS_ERR_FLASH_ERASE;
73}

75static int Flash_ProgramWords(uint32_t dst_addr, const uint32_t* words, uint32_t word_count)
76{

119static int ReadOne(uint32_t base, AppSettings* out)
120{
121 if (base == 0u) return SETTINGS_ERR_MAGIC_VERSION;
122 Flash_ReadWords(base, (uint32_t*)out, (uint32_t)((sizeof(AppSettings) + 3u)/4u));
123
124 if (out->magic != SETTINGS_MAGIC || out->version != SETTINGS_VERSION)
125 return SETTINGS_ERR_MAGIC_VERSION;
126
127 uint32_t calc = crc32_calc(out, sizeof(AppSettings) - sizeof(uint32_t));
128 if (calc != out->crc32) return SETTINGS_ERR_CRC;
129
130 return SETTINGS_OK;
131}

133static int WriteOne(uint32_t base, const AppSettings* in)
134{
135 AppSettings tmp = *in;
136 tmp.magic = SETTINGS_MAGIC;
137 tmp.version = SETTINGS_VERSION;
138 tmp.crc32 = 0u;
139 tmp.crc32 = crc32_calc(&tmp, sizeof(AppSettings) - sizeof(uint32_t));
140
141 const uint32_t start = base;
142 const uint32_t end = base + (uint32_t)((sizeof(AppSettings) + 3u) & ~3u);
143
144 int er = Flash_EraseRange(start, end);
145 if (er != SETTINGS_OK) return er;
146
147 int pr = Flash_ProgramWords(base, (const uint32_t*)&tmp, (uint32_t)((sizeof(AppSettings) + 3u)/4u));
148 return pr;
149}

179void Settings_Init(void)
180{
181 PickBases(&g_baseA, &g_baseB);
182
183 AppSettings a = {0}, b = {0};
184 int ra = ReadOne(g_baseA, &a);
185 int rb = (g_baseB ? ReadOne(g_baseB, &b) : SETTINGS_ERR_MAGIC_VERSION);
186
187 if (ra == SETTINGS_OK && rb == SETTINGS_OK) {
188 if (b.gen > a.gen) { g_settings = b; g_usingB = 1; }
189 else { g_settings = a; g_usingB = 0; }
190 } else if (ra == SETTINGS_OK) {
191 g_settings = a; g_usingB = 0;
192 } else if (rb == SETTINGS_OK) {
193 g_settings = b; g_usingB = 1;
194 } else {
195 Settings_FillDefaults(&g_settings);
196 (void)Settings_Save();
197 }
198}

214 } else {
