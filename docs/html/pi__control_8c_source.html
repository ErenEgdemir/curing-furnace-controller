38float PI_Control_Update(PI_Oven* c, float setpoint, float measurement, float dt)
39{
40 /* 1) Ölçümü filtrele (EMA) */
41 float y = measurement;
42 if (c->meas_alpha < 1.0f){
43 float a = c->meas_alpha; if (a < 0.0f) a = 0.0f; if (a > 1.0f) a = 1.0f;
44 c->y_filt = (c->y_filt == 0.0f) ? y : (a*y + (1.0f - a)*c->y_filt);
45 y = c->y_filt;
46 } else {
47 c->y_filt = y;
48 }
49
50 /* 2) Hata ve P terimi */

52 float up = c->kp * e;
53
54 /* 3) İ n t e g r a l (anti-windup ile) */
55 if (fabsf(e) > 10.0f) {
56
57 }else if(fabsf(e) < 7.0f){
58 if (dt > 0.0f && c->enabled){
59 c->integrator += c->ki * e * dt;
60 /* Çıkış sınırını aşmayacak şekilde önceden kısıtla */
