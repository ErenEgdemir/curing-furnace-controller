155static uint8_t u8x8_d_ssd1306_72x40_generic(u8x8_t *u8x8, uint8_t msg, uint8_t arg_int, void *arg_ptr)
156{
157 uint8_t x, c;
158 uint8_t *ptr;
159 switch(msg)
160 {
161 /* handled by the calling function
162 case U8X8_MSG_DISPLAY_SETUP_MEMORY:
163 u8x8_d_helper_display_setup_memory(u8x8, &u8x8_ssd1306_72x40_display_info);
164 break;
165 case U8X8_MSG_DISPLAY_INIT:
166 u8x8_d_helper_display_init(u8x8);
167 u8x8_cad_SendSequence(u8x8, u8x8_d_ssd1306_72x40_er_init_seq);
168 break;
169 */
170 case U8X8_MSG_DISPLAY_SET_POWER_SAVE:
171 if ( arg_int == 0 )
172 u8x8_cad_SendSequence(u8x8, u8x8_d_ssd1306_72x40_powersave0_seq);
173 else
174 u8x8_cad_SendSequence(u8x8, u8x8_d_ssd1306_72x40_powersave1_seq);
175 break;
176 case U8X8_MSG_DISPLAY_SET_FLIP_MODE:
177 if ( arg_int == 0 )
178 {
179 u8x8_cad_SendSequence(u8x8, u8x8_d_ssd1306_72x40_flip0_seq);
180 u8x8->x_offset = u8x8->display_info->default_x_offset;
181 }
182 else
183 {
184 u8x8_cad_SendSequence(u8x8, u8x8_d_ssd1306_72x40_flip1_seq);
185 u8x8->x_offset = u8x8->display_info->flipmode_x_offset;
186 }
187 break;
188#ifdef U8X8_WITH_SET_CONTRAST
189 case U8X8_MSG_DISPLAY_SET_CONTRAST:
190 u8x8_cad_StartTransfer(u8x8);
191 u8x8_cad_SendCmd(u8x8, 0x081 );
192 u8x8_cad_SendArg(u8x8, arg_int ); /* ssd1306 has range from 0 to 255 */
193 u8x8_cad_EndTransfer(u8x8);
194 break;
195#endif
196 case U8X8_MSG_DISPLAY_DRAW_TILE:
197 u8x8_cad_StartTransfer(u8x8);
198 x = ((u8x8_tile_t *)arg_ptr)->x_pos;
199 x *= 8;
200 x += u8x8->x_offset;
201 u8x8_cad_SendCmd(u8x8, 0x010 | (x>>4) );
202 u8x8_cad_SendCmd(u8x8, 0x000 | ((x&15)));
203 u8x8_cad_SendCmd(u8x8, 0x0b0 | (((u8x8_tile_t *)arg_ptr)->y_pos));
204
205 do
206 {
207 c = ((u8x8_tile_t *)arg_ptr)->cnt;
208 ptr = ((u8x8_tile_t *)arg_ptr)->tile_ptr;
209 u8x8_cad_SendData(u8x8, c*8, ptr); /* note: SendData can not handle more than 255 bytes */
210 arg_int--;
211 } while( arg_int > 0 );
212
213 u8x8_cad_EndTransfer(u8x8);
214 break;
215 default:
216 return 0;
217 }
218 return 1;
219}

250uint8_t u8x8_d_ssd1306_72x40_er(u8x8_t *u8x8, uint8_t msg, uint8_t arg_int, void *arg_ptr)
251{
252 if ( msg == U8X8_MSG_DISPLAY_SETUP_MEMORY )
253 {
254 u8x8_d_helper_display_setup_memory(u8x8, &u8x8_ssd1306_72x40_display_info);
255 return 1;
256 }
257 else if ( msg == U8X8_MSG_DISPLAY_INIT )
258 {
259 u8x8_d_helper_display_init(u8x8);
260 u8x8_cad_SendSequence(u8x8, u8x8_d_ssd1306_72x40_er_init_seq);
261 return 1;
262 }
263 return u8x8_d_ssd1306_72x40_generic(u8x8, msg, arg_int, arg_ptr);
264}
