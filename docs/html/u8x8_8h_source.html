36 U8glib has several layers. Each layer is implemented with a callback function.
37 This callback function handles the messages for the layer.
38
39 The topmost level is the display layer. It includes the following messages:
40
41 U8X8_MSG_DISPLAY_SETUP_MEMORY no communication with the display, setup memory only
42 U8X8_MSG_DISPLAY_INIT
43 U8X8_MSG_DISPLAY_SET_FLIP_MODE
44 U8X8_MSG_DISPLAY_SET_POWER_SAVE
45 U8X8_MSG_DISPLAY_SET_CONTRAST
46 U8X8_MSG_DISPLAY_DRAW_TILE
47
48 A display driver may decided to breakdown these messages to a lower level interface or
49 implement this functionality directly.
50
51
52 One layer is the Command/Arg/Data interface. It can be used by the display layer
53 to communicate with the display hardware.
54 This layer only deals with data, commands and arguments. D/C line is unknown.
55 U8X8_MSG_CAD_INIT
56 U8X8_MSG_CAD_SET_I2C_ADR (obsolete)
57 U8X8_MSG_CAD_SET_DEVICE (obsolete)
58 U8X8_MSG_CAD_START_TRANSFER
59 U8X8_MSG_CAD_SEND_CMD
60 U8X8_MSG_CAD_SEND_ARG
61 U8X8_MSG_CAD_SEND_DATA
62 U8X8_MSG_CAD_END_TRANSFER
63
64 The byte interface is there to send 1 byte (8 bits) to the display hardware.
65 This layer depends on the hardware of a microcontroller, if a specific hardware
66 should be used (I2C or SPI).
67 If this interface is implemented via software, it may use the GPIO level for sending
68 bytes.
69 U8X8_MSG_BYTE_INIT
70 U8X8_MSG_BYTE_SEND 30
71 U8X8_MSG_BYTE_SET_DC 31
72 U8X8_MSG_BYTE_START_TRANSFER
73 U8X8_MSG_BYTE_END_TRANSFER
74 U8X8_MSG_BYTE_SET_I2C_ADR (obsolete)
75 U8X8_MSG_BYTE_SET_DEVICE (obsolete)
76
77 GPIO and Delay
78 U8X8_MSG_GPIO_INIT
79 U8X8_MSG_DELAY_MILLI
80 U8X8_MSG_DELAY_10MICRO
81 U8X8_MSG_DELAY_100NANO
82 U8X8_MSG_DELAY_NANO
83*/

237struct u8x8_display_info_struct
238{
239 /* == general == */
240
241 uint8_t chip_enable_level; /* UC1601: 0 */
242 uint8_t chip_disable_level; /* opposite of chip_enable_level */
243
244 uint8_t post_chip_enable_wait_ns; /* UC1601: 5ns */
245 uint8_t pre_chip_disable_wait_ns; /* UC1601: 5ns */
246 uint8_t reset_pulse_width_ms; /* UC1601: 0.003ms --> 1ms */
247 uint8_t post_reset_wait_ms; /* UC1601: 6ms */
248
249
250 /* == SPI interface == */
251
252 /* after SDA has been applied, wait this much time for the SCK data takeover edge */
253 /* if this is smaller than sck_pulse_width_ns, then use the value from sck_pulse_width_ns */
254 uint8_t sda_setup_time_ns; /* UC1601: 12ns */
255 /* the pulse width of the the clock signal, cycle time is twice this value */
256 /* max freq is 1/(2*sck_pulse_width_ns) */
257 /* AVR: below 70: DIV2, 8 MHz, >= 70 --> 4MHz clock (DIV4) */
258 uint8_t sck_pulse_width_ns; /* UC1701: 50ns */
259
260 /* until here we have 8 bytes (uint8_t). Newly introduced for SPI.beginTransaction */
261 uint32_t sck_clock_hz;
262
263 /* previous name "sck_takeover_edge" renamed to "spi_mode" */
264 /* bit 0 of spi_mode is equal to the value of the previous variable sck_takeover_edge, 20 Aug 16: This is wrong the bit is actually inverted */
265 /* SPI has four clock modes: */
266 /* 0: clock active high, data out on falling edge, clock default value is zero, takover on rising edge */
267 /* 1: clock active high, data out on rising edge, clock default value is zero, takover on falling edge */
268 /* 2: clock active low, data out on rising edge */
269 /* 3: clock active low, data out on falling edge */
270 /* most displays have clock mode 1 */
271 uint8_t spi_mode;
272
273 /* == I2C == */
274 uint8_t i2c_bus_clock_100kHz; /* UC1601: 1000000000/275 = 37 *100k */
275
276
277 /* == 8 bit interface == */
278
279 /* how long to wait after all data line are set */
280 uint8_t data_setup_time_ns; /* UC1601: 30ns */
281 /* write enable pulse width */
282 uint8_t write_pulse_width_ns; /* UC1601: 40ns */
283
284 /* == layout == */
285 uint8_t tile_width;
286 uint8_t tile_height;
287
288 uint8_t default_x_offset; /* default x offset for the display */
289 uint8_t flipmode_x_offset; /* x offset, if flip mode is enabled */
290
291 /* pixel width is not used by the u8x8 procedures */
292 /* instead it will be used by the u8g2 procedures, because the pixel dimension can */
293 /* not always be calculated from the tile_width/_height */
294 /* the following conditions must be true: */
295 /* pixel_width <= tile_width*8 */
296 /* pixel_height <= tile_height*8 */
297 uint16_t pixel_width;
298 uint16_t pixel_height;
299};

478 Args: arg_int: 0: normal mode, 1: flipped HW screen (180 degree)
479 Tasks:
480 Reprograms the display controller to rotate the display by
481 180 degree (arg_int = 1) or not (arg_int = 0)
482 This may change u8g2->x_offset if the display is smaller than the controller ram
483 This message should only be supported if U8X8_WITH_FLIP_MODE is defined.

490/*
491 Name: U8X8_MSG_DISPLAY_DRAW_TILE
492 Args:
493 arg_int: How often to repeat this tile pattern
494 arg_ptr: pointer to u8x8_tile_t
495 uint8_t *tile_ptr; pointer to one or more tiles (number is "cnt")
496 uint8_t cnt; number of tiles
497 uint8_t x_pos; first tile x position
498 uint8_t y_pos; first tile y position
499 Tasks:
500 One tile has exactly 8 bytes (8x8 pixel monochrome bitmap).
501 The lowest bit of the first byte is the upper left corner
502 The highest bit of the first byte is the lower left corner
503 The lowest bit of the last byte is the upper right corner
504 The highest bit of the last byte is the lower left corner
505 "tile_ptr" is the address of a memory area, which contains
506 one or more tiles. "cnt" will contain the exact number of
507 tiles in the memory areay. The size of the memory area is 8*cnt;
508 Multiple tiles in the memory area form a horizontal sequence, this
509 means the first tile is drawn at x_pos/y_pos, the second tile is drawn
510 at x_pos+1/y_pos, third at x_pos+2/y_pos.
511 "arg_int" tells how often the tile sequence should be repeated:
512 For example if "cnt" is two and tile_ptr points to tiles A and B,
513 then for arg_int = 3, the following tile sequence will be drawn:
514 ABABAB. Totally, cnt*arg_int tiles will be drawn.
515
516*/
