42void DS1307_SetRegByte(uint8_t regAddr, uint8_t val) {
43 uint8_t bytes[2] = { regAddr, val };
44 HAL_I2C_Master_Transmit(_ds1307_ui2c, DS1307_I2C_ADDR << 1, bytes, 2, DS1307_TIMEOUT);
45}

52uint8_t DS1307_GetRegByte(uint8_t regAddr) {
53 uint8_t val;
54 HAL_I2C_Master_Transmit(_ds1307_ui2c, DS1307_I2C_ADDR << 1, &regAddr, 1, DS1307_TIMEOUT);
55 HAL_I2C_Master_Receive(_ds1307_ui2c, DS1307_I2C_ADDR << 1, &val, 1, DS1307_TIMEOUT);
56 return val;
57}
